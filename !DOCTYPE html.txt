function addDragScroll() {
  const filmStrip = document.getElementById('filmStrip');
  if(filmStrip) {
    let isDown = false, startX, scrollLeft;
    filmStrip.addEventListener('mousedown', (e) => { isDown = true; startX = e.pageX - filmStrip.offsetLeft; scrollLeft = filmStrip.scrollLeft; filmStrip.style.cursor = 'grabbing'; });
    window.addEventListener('mouseup', () => { isDown = false; if(filmStrip) filmStrip.style.cursor = 'grab'; });
    window.addEventListener('mousemove', (e) => { if(!isDown) return; e.preventDefault(); const x = e.pageX - filmStrip.offsetLeft; const walk = (x - startX) * 1.5; filmStrip.scrollLeft = scrollLeft - walk; });
  }
  const polTrack = document.querySelector('.pol-track');
  if(polTrack){
    let pDown=false,pStart,pLeft;
    polTrack.addEventListener('mousedown',(e)=>{ pDown=true; pStart=e.pageX-polTrack.offsetLeft; pLeft=polTrack.scrollLeft; polTrack.style.cursor='grabbing'; });
    window.addEventListener('mouseup',()=>{ pDown=false; if(polTrack) polTrack.style.cursor='grab'; });
    window.addEventListener('mousemove',(e)=>{ if(!pDown) return; e.preventDefault(); const walk=(e.pageX-polTrack.offsetLeft-pStart)*1.5; polTrack.scrollLeft=pLeft-walk; });
  }
}
